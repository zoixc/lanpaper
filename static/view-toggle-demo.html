// JavaScript
viewToggle.addEventListener('click', () => {
    const view = viewToggle.getAttribute('data-view');
    const newView = view === 'list' ? 'grid' : 'list';
    viewToggle.setAttribute('data-view', newView);
    // CSS animates automatically!
});
